/** * ガチャを実行する関数 * @param {number} times - 引く回数 * @returns {string[]} - 当たったレアリティの配列 */ function drawGacha(times = 1) { const rarities = [ { name: 'SSR', weight: 3 }, { name: 'SR', weight: 12 }, { name: 'R', weight: 35 }, { name: 'N', weight: 50 } ]; const results = []; for (let i = 0; i < times; i++) { // 0から99までの乱数を生成 const randomNum = Math.random() * 100; let cumulativeWeight = 0; for (const item of rarities) { cumulativeWeight += item.weight; if (randomNum < cumulativeWeight) { results.push(item.name); break; } } } return results; } // --- 実行 --- const tenPull = drawGacha(10); console.log("--- ガチャ結果 ---"); tenPull.forEach((item, index) => { console.log(`${index + 1}回目: [${item}]`); }); // SSRが含まれているかチェック if (tenPull.includes('SSR')) { console.log("\n🎉 おめでとう！SSRを引きました！"); } else { console.log("\n残念、SSRは出ませんでした。"); }
